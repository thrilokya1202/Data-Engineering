#include<stdio.h>
void sort(int a[30],int n)
{
        int temp;
        for(int i=0;i<n-1;i++)
        {
                for(int j=0;j<n-i-1;j++)
                {
                        if(a[j]>a[j+1])
                        {
                                temp=a[j];
                                a[j]=a[j+1];
                                a[j+1]=temp;
                        }
                }
        }
}
void main()
{
        int n,a[30];
        float median;
        printf("enter number of elements");
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
                printf("enter elements");
                scanf("%d",&a[i]);
        }
        sort(a,n);
        if(n%2==0)
        {
                median=(a[(n/2)-1]+a[n/2])/2.0;
        }
        else
        {
                median=a[n/2];
        }
        printf("median is %f",median);
}
